		$\leq 10^3$ & $O(n^2)$ & e.g. Bubble/Selection/Insertion Sort\\
		\hline
		$\leq 10^5$ & $O(n log_2 n)$ & e.g. Merge Sort, building Segment Tree\\
		\hline
		$\leq 10^6$ & $O(n),O(log_2n),O(1)$ & usually, contest problem has $n\leq 10^6$(e.g. to read input)\\

	\end{tabular}
